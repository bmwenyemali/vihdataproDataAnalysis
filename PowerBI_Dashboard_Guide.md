## SECTION 1: POWER QUERY (M LANGUAGE) - DATA TRANSFORMATION

================================================================================

### 1.1 Main Data Table Load

```
let
    Source = Excel.Workbook(File.Contents("datavih.xlsx"), null, true),
    Sheet1 = Source{[Item="Sheet1",Kind="Sheet"]}[Data],
    PromotedHeaders = Table.PromoteHeaders(Sheet1, [PromoteAllScalars=true]),
    ChangedTypes = Table.TransformColumnTypes(PromotedHeaders,{
        {"provinces", type text},
        {"annees", Int64.Type},
        {"trimestres", type text},
        {"indicateurs", type text},
        {"cibles", type text},
        {"sexes", type text},
        {"tranches_ages", type text},
        {"Valeur", Int64.Type}
    }),
    RemovedNulls = Table.SelectRows(ChangedTypes, each [Valeur] <> null),
    ReplacedNullText = Table.ReplaceValue(ChangedTypes, null, "Non specifie", Replacer.ReplaceValue, {"sexes", "tranches_ages", "cibles"})
in
    ReplacedNullText
```

### 1.2 Date Table (for time intelligence)

```
let
    StartYear = 2020,
    EndYear = 2024,
    Source = List.Dates(#date(StartYear, 1, 1), Duration.Days(#date(EndYear, 12, 31) - #date(StartYear, 1, 1)) + 1, #duration(1, 0, 0, 0)),
    TableFromList = Table.FromList(Source, Splitter.SplitByNothing()),
    RenamedColumns = Table.RenameColumns(TableFromList,{{"Column1", "Date"}}),
    ChangedType = Table.TransformColumnTypes(RenamedColumns,{{"Date", type date}}),
    AddedYear = Table.AddColumn(ChangedType, "Year", each Date.Year([Date]), Int64.Type),
    AddedQuarter = Table.AddColumn(AddedYear, "Quarter", each "T" & Text.From(Date.QuarterOfYear([Date])), type text),
    AddedMonth = Table.AddColumn(AddedQuarter, "Month", each Date.Month([Date]), Int64.Type),
    AddedYearQuarter = Table.AddColumn(AddedMonth, "YearQuarter", each Text.From([Year]) & "-" & [Quarter], type text)
in
    AddedYearQuarter
```

### 1.3 Province Lookup Table

```
let
    Source = Table.Distinct(Table.SelectColumns(VIH_Data, {"provinces"})),
    AddedIndex = Table.AddIndexColumn(Source, "ProvinceID", 1, 1, Int64.Type),
    AddedRegion = Table.AddColumn(AddedIndex, "Region", each
        if [provinces] = "Kinshasa" then "Capitale"
        else if List.Contains({"Nord-Kivu", "Sud Kivu", "Ituri", "Maniema", "Tanganyika"}, [provinces]) then "Est"
        else if List.Contains({"Haut-Katanga", "Lualaba", "Haut-Lomami", "Lomami"}, [provinces]) then "Sud"
        else if List.Contains({"Kongo-Central", "Kwango", "Kwilu", "Mai-Ndombe"}, [provinces]) then "Ouest"
        else "Nord/Centre", type text)
in
    AddedRegion
```

## SECTION 2: DAX MEASURES FOR UNAIDS 95-95-95 CASCADE

================================================================================

### 2.1 Core Measures

```dax
-- Total Value Measure
Total Value = SUM(VIH_Data[Valeur])

-- Total Records
Total Records = COUNTROWS(VIH_Data)

-- Average Value
Average Value = AVERAGE(VIH_Data[Valeur])
```

### 2.2 UNAIDS First 95 - Testing/Know Status

```dax
-- Total Tested
Total Tested =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre de clients testés"
)

-- Total Diagnosed HIV+
Total Diagnosed =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre de clients diagnostiqués VIH+"
)

-- Positivity Rate
Positivity Rate =
DIVIDE([Total Diagnosed], [Total Tested], 0) * 100
```

### 2.3 UNAIDS Second 95 - Treatment

```dax
-- Total on Antiretroviral Treatment (TAR)
Total on TAR =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre de PVVIH sous TAR"
)

-- Treatment Coverage Rate (2nd 95)
Treatment Rate =
DIVIDE([Total on TAR], [Total Diagnosed], 0) * 100

-- Treatment Target (95%)
Treatment Target = 95
```

### 2.4 UNAIDS Third 95 - Viral Suppression

```dax
-- Total with Viral Suppression
Total Viral Suppressed =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre  de PVVIH sous TAR qui ont supprimée la charge virale"
)

-- Viral Suppression Rate (3rd 95)
Viral Suppression Rate =
DIVIDE([Total Viral Suppressed], [Total on TAR], 0) * 100

-- Suppression Target (95%)
Suppression Target = 95
```

### 2.5 Year-over-Year Measures

```dax
-- Previous Year Value
Previous Year Value =
CALCULATE(
    [Total Value],
    SAMEPERIODLASTYEAR('Date'[Date])
)

-- Year-over-Year Growth %
YoY Growth % =
VAR CurrentYear = [Total Value]
VAR PrevYear = [Previous Year Value]
RETURN
IF(
    ISBLANK(PrevYear),
    BLANK(),
    DIVIDE(CurrentYear - PrevYear, PrevYear, 0) * 100
)

-- YoY Change
YoY Change = [Total Value] - [Previous Year Value]
```

### 2.6 Ranking Measures

```dax
-- Province Rank
Province Rank =
RANKX(
    ALL(VIH_Data[provinces]),
    [Total Value],
    ,
    DESC,
    Dense
)

-- Province Rank for Selected Year
Province Rank Selected =
RANKX(
    ALLSELECTED(VIH_Data[provinces]),
    [Total Value],
    ,
    DESC,
    Dense
)
```

### 2.7 KPI Indicators

```dax
-- Testing KPI Status
Testing KPI =
VAR TestRate = [Total Tested]
RETURN
IF(TestRate > 0, "Active", "No Data")

-- Treatment KPI Status
Treatment KPI Status =
VAR Rate = [Treatment Rate]
RETURN
SWITCH(
    TRUE(),
    Rate >= 95, "Target Met",
    Rate >= 80, "On Track",
    Rate >= 50, "Needs Improvement",
    "Critical"
)

-- Viral Suppression KPI Status
Suppression KPI Status =
VAR Rate = [Viral Suppression Rate]
RETURN
SWITCH(
    TRUE(),
    Rate >= 95, "Target Met",
    Rate >= 80, "On Track",
    Rate >= 50, "Needs Improvement",
    "Critical"
)
```

### 2.8 Prevention Measures

```dax
-- Male Condoms Distributed
Male Condoms =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre de préservatifs masculins distribués"
)

-- Female Condoms Distributed
Female Condoms =
CALCULATE(
    SUM(VIH_Data[Valeur]),
    VIH_Data[indicateurs] = "Nombre de préservatifs féminins distribués"
)

-- Total Prevention (Condoms)
Total Condoms = [Male Condoms] + [Female Condoms]
```

## SECTION 8: STEP-BY-STEP CREATION GUIDE

================================================================================

### Step 1: Open Power BI Desktop

1. Click "Get Data"
2. Select "Excel" or "Text/CSV"
3. Navigate to datavih_cleaned.csv
4. Click "Load" (or Transform Data first)

### Step 2: Create Measures

1. Go to Model view
2. Right-click on VIH_Data table
3. Select "New Measure"
4. Enter each DAX formula from Section 2

### Step 3: Create Date Table

1. Go to Model view
2. Click "New Table"
3. Enter the Date table DAX or use Power Query

### Step 4: Build Relationships

1. Go to Model view
2. Drag annees to Year
3. Drag trimestres to Quarter

### Step 5: Create Pages

1. Click "+" at bottom to add new pages
2. Rename pages according to Section 3

### Step 6: Add Visuals

1. Select visual type from Visualizations pane
2. Drag fields to Values, Axis, Legend
3. Apply formatting from Section 5

### Step 7: Create Bookmarks

1. Go to View > Bookmarks
2. Set up each page state
3. Add bookmark

### Step 8: Add Navigation

1. Insert > Buttons > Blank
2. Format button text
3. Set Action to Bookmark

### Step 9: Publish

1. Go to Home > Publish
2. Select workspace
3. Share dashboard link
